import 'package:flutter/material.dart';
import '../../../../core/theme/app_colors.dart';

class CategoryIconBadge extends StatelessWidget {
  final String category;

  const CategoryIconBadge({super.key, required this.category});

  @override
  Widget build(BuildContext context) {
    IconData icon;
    Color color;

    switch (category) {
      case 'Food':
        icon = Icons.restaurant;
        color = AppColors.food;
        break;
      case 'Transport':
        icon = Icons.directions_car;
        color = AppColors.transport;
        break;
      default:
        icon = Icons.shopping_bag;
        color = AppColors.shopping;
    }

    return Container(
      padding: const EdgeInsets.all(12),
      decoration: BoxDecoration(
        color: color.withOpacity(0.2),
        borderRadius: BorderRadius.circular(12),
      ),
      child: Icon(icon, color: color),
    );
  }
}

class HomeScreen extends StatefulWidget {
  const HomeScreen({super.key});

  @override
  State<HomeScreen> createState() => _HomeScreenState();
}

class _HomeScreenState extends State<HomeScreen> {
  bool isOnline = true;
  final TextEditingController _controller = TextEditingController();

  String _calculateInsight() {
    // mock calculation
    double thisWeek = 850;
    double lastWeek = 720;

    double difference = thisWeek - lastWeek;
    double percent = (difference / lastWeek) * 100;

    if (difference > 0) {
      return "You're spending ${percent.toStringAsFixed(0)}% more than last week.";
    } else {
      return "Great job! You spent ${percent.abs().toStringAsFixed(0)}% less than last week.";
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: AppColors.background,
      body: SafeArea(
        child: Padding(
          padding: const EdgeInsets.all(20),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              /// HEADER
              Text("BetterSpent", style: AppTextStyles.heading),
              const SizedBox(height: 6),
              Row(
                children: [
                  Text(
                    isOnline ? "ONLINE — SMART INPUT" : "OFFLINE — MANUAL MODE",
                    style: AppTextStyles.small.copyWith(
                      color: isOnline ? AppColors.primary : Colors.grey,
                    ),
                  ),
                  const Spacer(),
                  Switch(
                    value: isOnline,
                    activeColor: AppColors.primary,
                    onChanged: (value) {
                      setState(() {
                        isOnline = value;
                      });
                    },
                  ),
                ],
              ),

              const SizedBox(height: 24),

              /// STAT GRID
              HomeStatGrid(
                balance: "\$4,250",
                income: "\$5,000",
                maxWeek: "\$1,200",
                maxDay: "\$170",
              ),

              const SizedBox(height: 16),

              /// INSIGHT (always shown, calculated locally)
              SpendingInsightCard(message: _calculateInsight()),

              const SizedBox(height: 16),

              /// EXPENSE INPUT SECTION
              if (isOnline)
                PrimaryCard(
                  child: Column(
                    children: [
                      TextField(
                        controller: _controller,
                        style: AppTextStyles.body,
                        decoration: const InputDecoration(
                          hintText: "Enter expense (e.g. 50 dinner)",
                          border: InputBorder.none,
                        ),
                      ),
                      const SizedBox(height: 12),
                      SizedBox(
                        width: double.infinity,
                        child: ElevatedButton(
                          style: ElevatedButton.styleFrom(
                            backgroundColor: AppColors.primary,
                            padding: const EdgeInsets.symmetric(vertical: 14),
                            shape: RoundedRectangleBorder(
                              borderRadius: BorderRadius.circular(12),
                            ),
                          ),
                          onPressed: () {
                            _controller.clear();
                          },
                          child: const Text("Add Expense"),
                        ),
                      ),
                    ],
                  ),
                )
              else
                PrimaryCard(
                  child: Column(
                    children: [
                      const Text(
                        "Smart input unavailable offline.",
                        style: AppTextStyles.body,
                      ),
                      const SizedBox(height: 12),
                      SizedBox(
                        width: double.infinity,

                          style: ElevatedButton.styleFrom(

class HomeStatGrid extends StatelessWidget {
  final String balance;
  final String income;
  final String maxWeek;
  final String maxDay;

  const HomeStatGrid({
    super.key,
    required this.balance,
    required this.income,
    required this.maxWeek,
    required this.maxDay,
  });

  Widget _buildItem(String title, String value) {
    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        Text(title, style: AppTextStyles.small),
        const SizedBox(height: 6),
        Text(value, style: AppTextStyles.heading),
      ],
    );
  }

  @override
  Widget build(BuildContext context) {
    return PrimaryCard(
      child: Column(
        children: [
          Row(
            children: [
              Expanded(child: _buildItem("Balance", balance)),
              Expanded(child: _buildItem("Income", income)),
            ],

          const SizedBox(height: 16),
          Row(
            children: [
              Expanded(child: _buildItem("Max / Week", maxWeek)),
              Expanded(child: _buildItem("Max / Day", maxDay)),
            ],
          ),
        ],
      ),
    );
  }
